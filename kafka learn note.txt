tar -xzf kafka_2.10-0.8.2.0.tgz
cd kafka_2.10-0.8.2.0

bin/zookeeper-server-start.sh config/zookeeper.properties

bin/kafka-server-start.sh config/server.properties

不使用kafka自带zk，修改server.properties中的
zookeeper.connect=localhost:2181=>zookeeper.connect=10.10.1.4\:2181/kafka

kafka都log文件设置
log.dirs=/tmp/kafka1-logs,/tmp/kafka2-logs

Create a topic
bin/kafka-topics.sh --create --zookeeper localhost:2181 --replication-factor 1 --partitions 1 --topic test

check topic list
bin/kafka-topics.sh --list --zookeeper localhost:2181

Send some messages
bin/kafka-console-producer.sh --broker-list localhost:9092 --topic test 
check Received messages 
bin/kafka-console-consumer.sh --zookeeper localhost:2181 --topic test --from-beginning

performance test

bin/kafka-producer-perf-test.sh --broker-list=localhost:9092 --messages 10000 --topic test --threads 10 --message-size 1000 --batch-size 100 --compression-codec 1

Setting up a multi-broker cluster
cp config/server.properties config/server-1.properties
cp config/server.properties config/server-2.properties

config/server-1.properties:
    broker.id=1
    port=9093
    log.dir=/tmp/kafka-logs-1
 
config/server-2.properties:
    broker.id=2
    port=9094
    log.dir=/tmp/kafka-logs-2
